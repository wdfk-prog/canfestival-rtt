146static UNS8 block_query_BIT_change(UNS16 *value,UNS8 bit,uint16_t timeout,uint16_t block_time)
147{
148 uint16_t cnt = 0;
149
150 while(!CANOPEN_GET_BIT(*value,bit))
151 {
152 if((cnt * block_time) < timeout)
153 {
154 cnt++;
155 rt_thread_mdelay(block_time);
156 }
157 else
158 {
159 return 0xFF;
160 }
161 }
162 return 0x00;
163}

535static void motor_state(UNS8 nodeId)
536{
537 LOG_I("Mode operation:%d",*Modes_of_operation_Node[nodeId - 2].map_val);
538 LOG_I("ControlWord 0x%4.4X", *Controlword_Node[nodeId - 2].map_val);
539 LOG_I("StatusWord 0x%4.4X", *Statusword_Node[nodeId - 2].map_val);
540
541 if(CANOPEN_GET_BIT(*Statusword_Node[nodeId - 2].map_val , FAULT))
542 LOG_E("motor fault!");
543 if(CANOPEN_GET_BIT(*Statusword_Node[nodeId - 2].map_val , WARNING))
544 LOG_W("motor warning!");
545 if(CANOPEN_GET_BIT(*Statusword_Node[nodeId - 2].map_val , FOLLOWING_ERROR))
546 {
547 if(*Modes_of_operation_Node[nodeId - 2].map_val == PROFILE_POSITION_MODE)
548 LOG_E("motor following error!");
549 }
550 if(CANOPEN_GET_BIT(*Statusword_Node[nodeId - 2].map_val , POSITIVE_LIMIT))
551 LOG_W("motor touch positive limit!");
552 if(!CANOPEN_GET_BIT(*Statusword_Node[nodeId - 2].map_val , TARGET_REACHED))
553 LOG_W("The node did not receive the target arrival command!");
554
555 if(CANOPEN_GET_BIT(*Statusword_Node[nodeId - 2].map_val , 13))
556 {
557 if(*Modes_of_operation_Node[nodeId - 2].map_val == PROFILE_POSITION_MODE)
558 {
559 LOG_E("motor following error!");
560 }
561 else if(*Modes_of_operation_Node[nodeId - 2].map_val == HOMING_MODE)
562 {
563 LOG_E("motor Homing error occurred!");
564 }
565 }
566
567 LOG_I("current position %d PUU", *Position_actual_value_Node[nodeId - 2].map_val); //注意为0X6064，0X6063为增量式位置
568 LOG_I("current speed %.1f RPM", *Velocity_actual_value_Node[nodeId - 2].map_val / 10.0f);//注意单位为0.1rpm
569}
