459static void config_node_param_cb(CO_Data* d, UNS8 nodeId)
460{

488 conf->state++;
489 conf->try_cnt = 0;

499void config_node(uint8_t nodeId)
500{
501 if(Write_SLAVE_control_word(nodeId,0x80) == 0xFF)//初始化进行错误重置
502 {
503 LOG_E("nodeId:%d,Failed to clear error.The current node is not in operation",nodeId);
504 rt_sem_init(&(slave_conf[nodeId - 2].finish_sem), "servocnf", 0, RT_IPC_FLAG_FIFO);
505 if(rt_sem_take(&(slave_conf[nodeId - 2].finish_sem), SDO_REPLY_TIMEOUT) != RT_EOK)
506 {
507 slave_conf[nodeId - 2].err_code = NODEID_CONFIG_NO_RESPOND;
508 //掉线情况执行，重新上电情况无需处理
509 LOG_W("Waiting for the repair to complete, CAN communication is currently unavailable");
510 master402_fix_config_err(OD_Data,nodeId);
511 }
512 rt_sem_detach(&(slave_conf[nodeId - 2].finish_sem));
513 }
514 else
515 {
516 slave_conf[nodeId - 2].state = 0;
517 slave_conf[nodeId - 2].try_cnt = 0;
518 rt_sem_init(&(slave_conf[nodeId - 2].finish_sem), "servocnf", 0, RT_IPC_FLAG_FIFO);
519
520 EnterMutex();
521 LOG_I("The configuration starts for node %d",nodeId);
522 config_node_param(nodeId, &slave_conf[nodeId - 2]);
523 LeaveMutex();
524 rt_sem_take(&(slave_conf[nodeId - 2].finish_sem), RT_WAITING_FOREVER);
525 rt_sem_detach(&(slave_conf[nodeId - 2].finish_sem));
526
527 if(slave_conf[nodeId - 2].err_code != 0X00)//因配置错误导致的退出
528 {
529 LOG_E("Failed to configure the dictionary for node %d",nodeId);
530 if(slave_conf[nodeId - 2].err_code == NODEID_CONFIG_NO_SEND)
531 {
532 LOG_E("The configuration was not sent because the local dictionary failed");
533 }
534 else if(slave_conf[nodeId - 2].err_code == NODEID_CONFIG_NO_RESPOND)
535 {
536 LOG_E("The configuration reply did not respond, and the node dictionary failed");
537 }
538 LOG_W("Waiting for the repair to complete, CAN communication is currently unavailable");
539 master402_fix_config_err(OD_Data,nodeId);
540 return; //退出线程
541 }
542 else
543 {
544 UNS32 errorCode,map_val, size = SDO_MAX_LENGTH_TRANSFER;
545 UNS8 data_type;
546 errorCode = readLocalDict(OD_Data, 0x1016, nodeId - 1, &map_val, &size, &data_type, 0);
547 if(errorCode == OD_SUCCESSFUL)
548 {
551 UNS32 consumer_heartbeat_time = HEARTBEAT_FORMAT(nodeId,CONSUMER_HEARTBEAT_TIME);//写入节点的心跳时间
552 errorCode = writeLocalDict(OD_Data, 0x1016, nodeId - 1, &consumer_heartbeat_time, &size, 0);
553 if(errorCode != OD_SUCCESSFUL)
554 LOG_E("index:0X%04X,subIndex:0X%X,write Local Dict false,abort code is 0X%08X",0x1016,nodeId - 1,errorCode);
555 }
556 else
557 {
558 LOG_E("index:0X%04X,subIndex:0X%X,write Local Dict false,abort code is 0X%08X",0x1016,nodeId - 1,errorCode);
559 LOG_W("Node %d is not configured with a consumer heartbeat",nodeId);
560 }
561 //节点进入操作状态
562 masterSendNMTstateChange(OD_Data, nodeId, NMT_Start_Node);
563 LOG_I("Node %d configuration Complete",nodeId);
564 rt_thread_mdelay(200);//确保NMT命令下发成功
565 }
566 }

686static bool writeNetworkDictSync (CO_Data* d, UNS8 nodeId, UNS16 index,
687 UNS8 subIndex, UNS32 count, UNS8 dataType, void *data, UNS8 useBlockMode)
688{
689 if(nodeId < 1 || nodeId > MAX_NODE_COUNT - 1)
690 {
691 LOG_W("invalid nodeId:%d, should between 2 and %d",nodeId,MAX_NODE_COUNT-1);
692 return false;
693 }
694
695 int try_cnt = 3;
696 node_config_state* conf = &slave_conf[nodeId - 2];
697 rt_sem_init(&conf->finish_sem, "servocnf", 0, RT_IPC_FLAG_FIFO);
698
699 while(try_cnt--)
700 {
701 if(writeNetworkDictCallBack(d, nodeId, index, subIndex,
702 count, dataType, data, writeNetworkDictSyncCb, useBlockMode) != 0)
703 {
704 LOG_W("write SDO failed! nodeId = %d, index: 0X%04X, subIndex: 0X%X", nodeId, index, subIndex);
705 closeSDOtransfer(d, nodeId, SDO_CLIENT);
706 continue;
707 }
708
709 if(rt_sem_take(&conf->finish_sem, SDO_REPLY_TIMEOUT) != RT_EOK)
710 {
711 LOG_W("write SDO timeout! nodeId = %d, index: 0X%04X, subIndex: 0X%X", nodeId, index, subIndex);
712 closeSDOtransfer(d, nodeId, SDO_CLIENT);
713 continue;
714 }
715
716 UNS32 abortCode = 0;
717 UNS8 res = getWriteResultNetworkDict(d, nodeId, &abortCode);
718 closeSDOtransfer(d, nodeId, SDO_CLIENT);
719
720 if(res != SDO_FINISHED)
721 {
722 LOG_W("get SDO write result failed! nodeId = %d, index: 0X%04X, subIndex: 0X%X, abortCode = 0X%08X", nodeId, index, subIndex, abortCode);
723 continue;
724 }
725 rt_sem_detach(&conf->finish_sem);
726 return true;
727 }
728 rt_sem_detach(&conf->finish_sem);
729 return false;
730}
