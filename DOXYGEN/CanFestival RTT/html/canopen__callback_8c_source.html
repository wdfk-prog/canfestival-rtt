75 if(++node[heartbeatID - 2].try_cnt <= 5)

80 rt_thread_t tid;

89 LOG_E("canfestival get nodeid state thread start failed!");
90 }
91 else
92 {
93 rt_thread_startup(tid);
94 }
95 }
96 else
97 {
98 LOG_W("nodeID :%d,Unable to create a new thread because an existing thread is running",heartbeatID);
99 }
100 }
101 else if(node[heartbeatID - 2].try_cnt > 5)
102 {
103 LOG_E("NodeID:%d,The number %d of repairs is too many. It is confirmed that it is not an occasional anomaly. It will not be repaired any more.",heartbeatID,node[heartbeatID - 2].try_cnt);

247void master_resume_start(CO_Data *d,UNS8 nodeId)
248{
249 if(getState(d) == Stopped)//can通信异常，发送失败多次，进入停止状态
250 {
251 //Stop状态时会删除生产者心跳定时器
252 if (*d->ProducerHeartBeatTime)//恢复到OP状态时检查是否有生产者心跳时间
253 {
254 TIMEVAL time = *d->ProducerHeartBeatTime;
255 extern void ProducerHeartbeatAlarm(CO_Data* d, UNS32 id);
256 //设置生产者时间定时器，并设置定时回调
257 LOG_W("Restart the producer heartbeat");
258 d->ProducerHeartBeatTimer = SetAlarm(d, 0, &ProducerHeartbeatAlarm, MS_TO_TIMEVAL(time), MS_TO_TIMEVAL(time));
259 }
260 LOG_W("The master station enters the operation state from the stop state");
261 setState(d, Operational);//转入Operational状态
262 }

271static void master402_fix_node_Disconnected(void* parameter)
272{
273 static e_nodeState last;
274 e_nodeState now;
275
276 int heartbeatID = (int)parameter;//强制转换为16位数据与void*指针字节一致，以消除强制转换大小不匹配警告
277 LOG_E("heartbeatID abnormal:0x%x ", heartbeatID);
278 node[heartbeatID - 2].lock = 0xff;
279 while (1)
280 {
281 last = now;
282 now = getNodeState(OD_Data,heartbeatID);
283 if(now != last)//节点状态发生变化，证明从机上线
284 {
285 if(now == Operational)//由0x8130错误码处理程序处理
286 {
287 LOG_I("nodeID:%d,Handled by the 0x8130 error code handler,def ThreadFinished",heartbeatID);
288 node[heartbeatID - 2].lock = 0;
289 return;//删除线程
290 }
291 else if(now == Pre_operational)
292 {
293 master_resume_start(OD_Data,heartbeatID);
294 masterSendNMTstateChange(OD_Data,heartbeatID,NMT_Start_Node);
295 LOG_I("nodeID:%d,Determines that the line is restored and switches the slave machine to operation mode, deleting the current thread",heartbeatID);
296 node[heartbeatID - 2].lock = 0;
297 return;//退出线程
298 }
299 else if(now == Initialisation)
300 {
301// setState(OD_Data, Initialisation);//Initialisation
302 LOG_I("nodeID:%d,After the heartbeat of the node is abnormal, the node is shut down and powered on",heartbeatID);
303 node[heartbeatID - 2].lock = 0;
304 return;//退出线程
305 }
306 }
307
308 rt_thread_mdelay(CONSUMER_HEARTBEAT_TIME);//按照获取心跳间隔时间进行判断
309 }

317static void master402_fix_config_err_thread_entry(void* parameter)
318{
319 static e_nodeState last;
320 e_nodeState now;
321 int nodeId = (int)parameter;//强制转换为16位数据与void*指针字节一致，以消除强制转换大小不匹配警告
322 LOG_E("nodeId abnormal:0x%x ", nodeId);
323 while (1)
324 {
325 now = getNodeState(OD_Data,nodeId);
326 if(now == Unknown_state)//没有进行通信
327 {
328 masterRequestNodeState(OD_Data,nodeId);//发送Node Guarding Request命令查询节点状态
329 }
330 else if(now == Disconnected)//断开连接
331 {
332 masterRequestNodeState(OD_Data,nodeId);//发送Node Guarding Request命令查询节点状态
333 }
334 else if(now == Pre_operational)//通信恢复
335 {
336 master_resume_start(OD_Data,nodeId);
337 masterSendNMTstateChange(OD_Data,nodeId,NMT_Start_Node);
338 config_node(nodeId);
339 LOG_I("nodeID:%d,The line comm.unication of the node is restored",nodeId);
340 return; //删除线程
341 }
342 else if(now == Initialisation)//节点断电后上电
343 {
344 LOG_I("nodeID:%d,Restart the node after the node is shut down",nodeId);
345 return; //删除线程
346 }
347 else
348 {
349 masterRequestNodeState(OD_Data,nodeId);//发送Node Guarding Request命令查询节点状态
350 }
351
352 rt_thread_mdelay(CONSUMER_HEARTBEAT_TIME);//按照获取心跳间隔时间进行判断
353 }
