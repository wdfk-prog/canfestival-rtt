137void canopen_recv_thread_entry(void* parameter)
138{
139 struct can_app_struct *canpara = (struct can_app_struct *) parameter;
140 struct rt_can_msg msg;
141 Message co_msg;
142
143 candev = rt_device_find(canpara->name);
144 RT_ASSERT(candev);
145 rt_sem_init(&can_data.sem, "co-rx", 0, RT_IPC_FLAG_PRIO);
146 rt_err_t err = rt_device_open(candev, (RT_DEVICE_OFLAG_RDWR | RT_DEVICE_FLAG_INT_RX | RT_DEVICE_FLAG_INT_TX));
147 if( err != RT_EOK)
148 {
149 LOG_E("canfestival open device %s failed, err = %d", canpara->name, err);
150 return;
151 }
152 /* 设置 CAN 通信的波特率*/
153 err = rt_device_control(candev, RT_CAN_CMD_SET_BAUD, (void *)CAN_BAUD);
154 if( err != RT_EOK)
155 {
156 LOG_E("canfestival set baud failed, err = %d", err);
157 return;
158 }
159 err = rt_device_set_rx_indicate(candev, can1ind);
160 if( err != RT_EOK)
161 {
162 LOG_E("canfestival set rx indicate failed, err = %d", err);
163 return;
164 }
165
166 rt_size_t read_size = 0;
167 while (1)
168 {
169 err = rt_sem_take(&can_data.sem, MAX_SEM_WAIT_TIME);
170 if ( err != RT_EOK)
171 {
172 if(getState(od_data) == Operational)
173 {
174 LOG_W("canfestival wait receive timeout, err = %d", err);
175 }
176 }
177 else
178 {
179 read_size = rt_device_read(candev, 0, &msg, sizeof(msg));
180 if( read_size == sizeof(msg))
181 {
182 co_msg.cob_id = msg.id;
183 co_msg.len = msg.len;
184 co_msg.rtr = msg.rtr;
185 memcpy(co_msg.data, msg.data, msg.len);
186 EnterMutex();
187 canDispatch(od_data, &co_msg);
188 LeaveMutex();
189 }
190 else if (read_size == 0)
191 {
192 LOG_W("canfestival receive faild, err = %d", rt_get_errno());
193 }
194 else
195 {
196 LOG_W("canfestival receive size wrong, size = %u", read_size);
197 }
198 }
199 }
200}
