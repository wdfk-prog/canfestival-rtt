53void setTimer(TIMEVAL value)
54{
55 rt_hwtimerval_t val;
56
57 val.sec = value / 1000000;
58 val.usec = value % 1000000;
61 if(val.usec < MIN_TIMER_TIMEOUT_US){
62 val.usec = MIN_TIMER_TIMEOUT_US;
63 }

67 if(rt_device_write(canfstvl_timer_dev, 0, &val, sizeof(val)) == 0) {
68 LOG_E("CANopen set timer failed, err = %d", rt_get_errno());
69 }

91static void canopen_timer_thread_entry(void* parameter)
92{
93 while(1)
94 {
95 if(rt_sem_take(canfstvl_timer_sem, RT_WAITING_FOREVER) != RT_EOK)
96 {
97 LOG_E("canfestival take timer sem failed");
98 return;
99 }
100
101 EnterMutex();
102 rt_size_t read_size = rt_device_read(canfstvl_timer_dev, 0, &last_timer_val, sizeof(last_timer_val));
103 if( read_size == 0)
104 {
105 LOG_E("canfestival read timer failed, err = %d", rt_get_errno());
106 }
107 else
108 {
109 TimeDispatch();
110 }
111 LeaveMutex();
112 }

132void initTimer(void)
133{
134 rt_thread_t tid;
135 rt_err_t err;
136 rt_hwtimer_mode_t mode;
137 int freq = 1000000;
138
139 canfstvl_timer_sem = rt_sem_create("canfstvl", 0, RT_IPC_FLAG_PRIO);
140
141 canfstvl_timer_dev = rt_device_find(CANFESTIVAL_TIMER_DEVICE_NAME);
142 RT_ASSERT(canfstvl_timer_dev != RT_NULL);
143 err = rt_device_open(canfstvl_timer_dev, RT_DEVICE_OFLAG_RDWR);
144 if (err != RT_EOK)
145 {
146 rt_kprintf("CanFestival open timer Failed! err=%d\n", err);
147 return;
148 }
149 rt_device_set_rx_indicate(canfstvl_timer_dev, timer_timeout_cb);
150 err = rt_device_control(canfstvl_timer_dev, HWTIMER_CTRL_FREQ_SET, &freq);
151 if (err != RT_EOK)
152 {
153 rt_kprintf("Set Freq=%dhz Failed\n", freq);
154 }
155
156 mode = HWTIMER_MODE_ONESHOT;
157 err = rt_device_control(canfstvl_timer_dev, HWTIMER_CTRL_MODE_SET, &mode);
158 rt_device_read(canfstvl_timer_dev, 0, &last_timer_val, sizeof(last_timer_val));
159#ifdef IRQ_PRIORITY
160 struct stm32_hwtimer *tim_device = RT_NULL;
161 tim_device = rt_container_of(canfstvl_timer_dev, struct stm32_hwtimer, time_device);
162
163 HAL_NVIC_SetPriority(tim_device->tim_irqn,IRQ_PRIORITY, 0);
164#endif
165 tid = rt_thread_create("cf_timer",
166 canopen_timer_thread_entry, RT_NULL,
167 THREAD_STACK_SIZE, THREAD_PRIORITY, THREAD_TIMESLICE);
168 if (tid != RT_NULL) rt_thread_startup(tid);
169
170}
